func lookupPlugin(p, s string) plugin.Symbol {
        plug, err := plugin.Open(p)
        errorHandler(err)

        sym, err := plug.Lookup(s)
        errorHandler(err)
        return sym
}

func errorHandler(err error) {
        if err != nil {
                fmt.Println(err)
                os.Exit(1)
        }
}
