class Pet
	attr_reader :color, :breed
	attr_accessor :name
	def initialize(color, breed)
		@color = color
		@breed = breed
		@hungry = true
	end
	def feed(food)
		puts "Mmmm, " + food + "!"
		@hungry = false
	end
	def hungry?
		if @hungry
			puts "I\'m hungry!"
		else
			puts "I\'m full!"
		end
		@hungry
	end
end

class Dog < Pet
	def speak
		puts "Woof!"
	end
end

puppy = Dog.new("black", "Stafforshire Terrier")
puppy.speak
puts puppy.breed

class Cat < Pet
	def speak
		puts "Meow!"
	end
end

kitty = Cat.new("grey", "Persian")

puts "Let\'s inspect our new cat:"
puts kitty.inspect
puts "What class does our new cat belong to?"

kitty.name = "Betsy"

puts "Is our cat hungry now?"
